\newcommand{\ffanotea}{%
	If the Active Player's first Casting Attempt is successful (including Miscasts), roll a D6. On a roll of 1--3, it is considered failed instead (and any Miscast is ignored), and all of the Magic Dice used for the Casting Attempt are returned to the pool.%
}

\newcommand{\ffanoteb}{%
	The Active Player cannot cast the Hereditary Spell of their army (this does not apply to Attribute Spells).%
}

\newcommand{\ffanotec}{%
	The Active Player cannot cast spells with type Hex unless they also have type Damage.%
}

\newcommand{\ffanoted}{%
	The Active Player cannot cast spells whose types include both Hex and Damage.%
}

\newcommand{\ffanotee}{%
	The Active Player cannot cast spells with type Augment.%
}

\newcommand{\ffanotef}{%
	The Active Player cannot cast spells with type Universal.%
}

\newcommand{\ffanoteg}{%
	Bound Spells cannot be cast with 2 Magic Dice.%
}

\newcommand{\ffanoteh}{%
	The Active Player may remove up to 15 Veil Tokens from its pool and generate up to 5 Magic Dice from Veil Tokens.%
}
