\newcommand{\watcherswoerestriction}{Cannot be taken by Wizards}
\newcommand{\watcherswoedef}{%
	Shots \textbf{always} set to 4. Shooting Attacks made with this weapon become \textbf{Poison Attacks}.
}
\newcommand{\boughofwyscandef}{%
	Shots \textbf{always} set to 1, Str 4 [6], AP 2 [10], \textbf{\areaattack{} (1\timess{}5)}, [\textbf{\multiplewounds{2}{}}], \textbf{\reload}.
}
\newcommand{\huntershonourdef}{%
	Attacks made with this weapon gain +1 Strength and +1 Armour Penetration. If the wielder causes at least one unsaved wound with this weapon, the wielder and all \rnf{} models in the wielder's unit gain \textbf{\distracting} until the end of the Melee Phase.%
}
\newcommand{\spiritofthewhirlwinddef}{%
	The wielder gains +1 Attack Value, and attacks made with this weapon gain +1 Strength and \textbf{\lethalstrike}.%
}
\newcommand{\oakenmightdef}{%
	Attacks made with this weapon gain +3 Strength.%
}

\newcommand{\shieldingbarkrestriction}{Infantry models only}
\newcommand{\shieldingbarkdef}{%
	The wearer gains +1 Armour, \textbf{\aegis{5}}, \textbf{\fearless}, \textbf{\flammable}, and \textbf{\magicalattacks}.%
}

\newcommand{\bannerofdeceptiondef}{%
	At the end of step 4 of the Deployment Phase Sequence (before deploying Scouts), the owner may remove the bearer's unit from the Battlefield and deploy it again elsewhere (any Characters joined to the unit must remain in the unit; this does not affect the number of Undeployed Units for calculating the starting roll-off bonus).%
}
\newcommand{\bannerofsilentmistdef}{%
	The bearer's unit gains Soft Cover. Enemy units within \distance{3} of the bearer's unit may not gain any benefit from a Musician.%
}
\newcommand{\predatorpennantdef}{%
	The bearer's unit gains \textbf{\devastatingcharge{\distracting}}.%
}

\newcommand{\hailshotrestriction}{Forest Princes and Chieftains only}
\newcommand{\hailshotdef}{%
	One use only. When this Artefact is used, it is a Shooting Weapon with the following profile:\newline
	\range{30}, \shots{3D6}, \St{} 4, \AP{} 1, \textbf{\magicalattacks}. Aim is \textbf{set} to 2+. When fired from Short Range, it gains +1 Armour Penetration. \masterarcher{} cannot be used in conjunction with \hailshot{}.%
}
\newcommand{\mistwalkersmirrorrestriction}{Models on foot only}
\newcommand{\mistwalkersmirrordef}{%
	One use only. May be activated at the end of  any friendly Movement Phase if the bearer’s unit consists entirely of Standard Height Infantry models, is unengaged, and is fully inside a Forest Terrain Feature that doesn’t contain any enemy models.%
	\ Apply the following rules:
	\begin{enumerate}
		\item The bearer's unit loses Scoring until the start of the next phase.
		\item Remove the bearer's unit from the Battlefield.
		\item Immediately place it back on the Battlefield  with Special Ambush (centre of a Forest Terrain Feature). If the unit cannot be placed, it is considered destroyed where it was removed.
	\end{enumerate}
}
\newcommand{\sacredseedsrestriction}{Models on foot only}
\newcommand{\sacredseedsdef}{%
	One use only. The player may activate this Artefact at the end of any friendly Movement Phase and place a Forest
	Terrain Feature in contact with the bearer and at least \distance{1} away from any enemy units and other Terrain Features. The Forest \textbf{must} fit within a circle with a diameter of \distance{6}.%
}
\newcommand{\drumsofcenyrnrestriction}{Models on foot only}
\newcommand{\drumsofcenyrndef}{%
	One use only. May be activated when the bearer's unit declares a Charge. The target of the Charge may only declare Hold as its Charge Reaction unless it is already Fleeing. The enemy unit may still declare Charge Reactions as normal if it is subsequently Charged by other units.%
}
\newcommand{\hornofthewildhuntdef}{%
	One use only. May be activated when a friendly unit within \distance{8} fails a roll for Charge Range. The roll may be rerolled.\columnbreak%
}
\newcommand{\glyphofamrylrestriction}{Cannot be taken by models with \sylvanspirit{}}
\newcommand{\glyphofamryldef}{%
	The bearer gains \textbf{Cannot be Stomped}. When fighting a Duel, the bearer gains +3 Defensive Skill.%
}
