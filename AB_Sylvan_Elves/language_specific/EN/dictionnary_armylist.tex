% Blade Dancers
\newcommand{\dancesofcenyrn}{Dances of Cenyrn}
\newcommand{\dancesofcenyrndef}{%
	At the start of each Round of Combat, units consisting entirely of models with this rule \textbf{must} choose one of the dances listed below and apply its effects until the end of the Round of Combat. The unit cannot choose this dance again until after one of the following has happened:%
	\begin{itemize}%
		\item The unit is no longer Engaged in Combat.%
		\item The unit has chosen a different dance.%
	\end{itemize}%
}

\newcommand{\dancesofcenyrntable}{%
	\startdancestable{}%
		\adddance{Dance of Whirling Blades}{+1 Attack Value.}%
		\adddance{Dance of the Parting Mists}{%
			\textbf{\aegis{} (3+)}, \minuss{}1 Strength, and \minuss{}1 Armour Penetration.%
		}%
		\adddance{Dance of Biting Wind}{+1 Armour Penetration and \textbf{\lethalstrike}.}%
		\adddance{Dance of Bedevilments}{%
			\textbf{\fear}, and enemy units in base contact with the model always have a Rank Bonus of 0.%
		}%
	\closedancestable{}%
}
